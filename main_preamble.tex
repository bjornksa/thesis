     \varwidth{\epigraphwidth}

     \epi@finalwidth=\wd\epi@sourcebox
   \fi

     \hb@xt@\epi@finalwidth{\hfil\box\epi@textbox}

     \vskip.75ex
     \hb@xt@\epi@finalwidth{\hfil\box\epi@sourcebox}
